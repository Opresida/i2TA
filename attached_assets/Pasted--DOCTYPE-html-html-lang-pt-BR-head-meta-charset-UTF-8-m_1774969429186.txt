    <!-- SCRIPT DE INTERATIVIDADE PARA ACENDER OS NÚMEROS -->
    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.stack-card');
            
            const handleScroll = () => {
                cards.forEach(card => {
                    // Pega a posição atual do cartão na tela
                    const rect = card.getBoundingClientRect();
                    // Pega o valor exato do 'top' configurado no CSS (ex: 100, 120, 140...)
                    const computedTop = parseFloat(window.getComputedStyle(card).top) || 0;
                    
                    // Se a distância do topo da tela for menor ou igual ao 'top' do CSS (com 2px de margem de erro)
                    // Significa que o cartão bateu no teto e EMPILHOU!
                    if (rect.top <= computedTop + 2) {
                        card.classList.add('is-stacked');
                    } else {
                        card.classList.remove('is-stacked');
                    }
                });
            };

            // Ouve a rolagem do mouse
            window.addEventListener('scroll', handleScroll, { passive: true });
            
            // Dispara uma vez ao carregar a página
            handleScroll();
        });
    </script>
